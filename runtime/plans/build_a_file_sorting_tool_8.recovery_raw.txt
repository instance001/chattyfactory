from pathlib import Path
def sort_files(src_dir: str, dst_dir: str):
    src = Path(src_dir)
    dst = Path(dst_dir)
    dst.mkdir(parents=True, exist_ok=True)
    for file in src.iterdir():
        if file.is_file():

            target = dst / ext
            target.mkdir(exist_ok=True)
            file.rename(target / file.name)
